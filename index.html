Kata from Codewars: (Sum of Digits) Given n, take the sum of the digits of n. If that value has more than one digit, continue reducing in this way until a single-digit number is produced. The input will be a non-negative integer. 


            
         function digitalRoot(n) {
            if (n >= 10) {
              const str = n.toString();
              const arr = str.split("");
              const arrNew = arr.map((el) => +el);
              const value = arrNew.reduce((el, acc) => {
              return +el + acc; },0);
              return digitalRoot(value);
             }
             return n;
            }
